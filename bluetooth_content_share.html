-- Table pending_actions (si pas déjà créée)
CREATE TABLE IF NOT EXISTS pending_actions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
  module_id UUID NOT NULL REFERENCES modules(id) ON DELETE CASCADE,
  action_type TEXT NOT NULL,
  reason TEXT,
  status TEXT DEFAULT 'pending',
  error_message TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  completed_at TIMESTAMPTZ
);

-- Trigger pour appeler process-action
CREATE OR REPLACE FUNCTION call_process_action_edge()
RETURNS TRIGGER
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
BEGIN
  PERFORM
    net.http_post(
      url := CONCAT(current_setting('app.settings.supabase_url'), '/functions/v1/process-action'),
      headers := jsonb_build_object(
        'Authorization', current_setting('app.settings.service_role_key'),
        'Content-Type', 'application/json'
      ),
      body := jsonb_build_object('record', row_to_json(NEW))
    );
  RETURN NEW;
END;
$$;

CREATE TRIGGER process_action_trigger
  AFTER INSERT ON pending_actions
  FOR EACH ROW
  EXECUTE FUNCTION call_process_action_edge();

-- Planification de orchestrate (toutes les 30 secondes)
-- Nécessite l'extension pg_cron
CREATE EXTENSION IF NOT EXISTS pg_cron;

SELECT cron.schedule(
  'orchestrate-every-30s',
  '*/30 * * * * *',
  $$
  SELECT
    net.http_post(

      headers := jsonb_build_object(
        'Authorization', current_setting('app.settings.service_role_key')
      )

  $$
);
