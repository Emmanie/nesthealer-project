// supabase/functions/process-batch-import/index.ts
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2.39.0';

Deno.serve(async (req) => {
  const { batchId, tenantId, sites } = await req.json();

  const supabase = createClient(
    Deno.env.get('SUPABASE_URL')!,
    Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
  );

  let succeeded = 0;
  let failed = 0;
  const errors: any[] = [];

  for (const site of sites) {
    // Vérifier si le site existe déjà (optionnel)
    const { data: existing } = await supabase
      .from('modules')
      .select('id')
      .eq('tenant_id', tenantId)
      .eq('url', site.url)
      .maybeSingle();

    if (existing) {
      failed++;
      errors.push({ site, error: 'URL already exists' });
      continue;
    }

    const { error } = await supabase.from('modules').insert({
      tenant_id: tenantId,
      name: site.name,
      url: site.url,
      config: site.config || {},
      status: 'active',
    });
    if (error) {
      failed++;
      errors.push({ site, error: error.message });
    } else {
      succeeded++;
    }
  }

  // Mettre à jour le batch
  await supabase
    .from('batch_imports')
    .update({
      status: 'completed',
      succeeded,
      failed,
      errors: errors.length ? errors : null,
      completed_at: new Date().toISOString(),
    })
    .eq('id', batchId);

  return new Response(JSON.stringify({ succeeded, failed }), { status: 200 });
});
