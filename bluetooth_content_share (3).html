// supabase/functions/process-action/index.ts
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2.39.0';

Deno.serve(async (req) => {
  const payload = await req.json();
  const { record } = payload; // déclenché par trigger

  const supabase = createClient(
    Deno.env.get('SUPABASE_URL')!,
    Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
  );

  const { id, tenant_id, module_id, action_type, reason } = record;

  try {
    let success = false;
    switch (action_type) {
      case 'repair':
        success = await repairModule(module_id, tenant_id, supabase);
        break;
      case 'restart':
        success = await restartModule(module_id, tenant_id, supabase);
        break;
      case 'rollback':
        success = await rollbackModule(module_id, tenant_id, supabase);
        break;
      case 'kill':
        success = await killModule(module_id, tenant_id, supabase);
        break;
      default:
        console.log(`Action inconnue: ${action_type}`);
    }

    // Mettre à jour le statut de l'action
    await supabase
      .from('pending_actions')
      .update({
        status: success ? 'completed' : 'failed',
        completed_at: new Date().toISOString(),
      })
      .eq('id', id);

    // Log dans action_logs
    await supabase.from('action_logs').insert({
      tenant_id,
      module_id,
      action: action_type,
      reason,
      success,
    });
  } catch (err) {
    await supabase
      .from('pending_actions')
      .update({
        status: 'failed',
        error_message: err.message,
        completed_at: new Date().toISOString(),
      })
      .eq('id', id);
  }

  return new Response(JSON.stringify({ success: true }), { status: 200 });
});

// Implémentations des actions
async function repairModule(moduleId: string, tenantId: string, supabase: any) {
  // Exemple : redémarrer le conteneur ou appeler une API externe
  // Ici on simule en réinitialisant les compteurs
  const { error } = await supabase
    .from('modules')
    .update({
      error_count: 0,
      status: 'active',
      last_success: new Date().toISOString(),
    })
    .eq('id', moduleId)
    .eq('tenant_id', tenantId);
  return !error;
}

async function restartModule(moduleId: string, tenantId: string, supabase: any) {
  // Redémarrage forcé
  const { error } = await supabase
    .from('modules')
    .update({
      error_count: 0,
      status: 'active',
      last_success: new Date().toISOString(),
    })
    .eq('id', moduleId)
    .eq('tenant_id', tenantId);
  return !error;
}

async function rollbackModule(moduleId: string, tenantId: string, supabase: any) {
  // Restaurer la configuration précédente
  const { data: module } = await supabase
    .from('modules')
    .select('previous_config, config')
    .eq('id', moduleId)
    .single();
  if (module && module.previous_config) {
    const { error } = await supabase
      .from('modules')
      .update({
        config: module.previous_config,
        previous_config: module.config,
        error_count: 0,
        status: 'active',
      })
      .eq('id', moduleId);
    return !error;
  }
  return false;
}

async function killModule(moduleId: string, tenantId: string, supabase: any) {
  // Désactiver le module
  const { error } = await supabase
    .from('modules')
    .update({ status: 'dead' })
    .eq('id', moduleId)
    .eq('tenant_id', tenantId);
  // Notifier l'admin (à implémenter)
  return !error;
}
