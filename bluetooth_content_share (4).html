// supabase/functions/orchestrate/index.ts
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2.39.0';

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Methods': 'POST, OPTIONS',
  'Access-Control-Allow-Headers': 'Content-Type, Authorization',
};

Deno.serve(async (req) => {
  if (req.method === 'OPTIONS') return new Response('ok', { headers: corsHeaders });

  const supabase = createClient(
    Deno.env.get('SUPABASE_URL')!,
    Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
  );

  // Récupérer tous les tenants actifs
  const { data: tenants, error: tenantsError } = await supabase
    .from('tenants')
    .select('id, plan, config');
  if (tenantsError) throw tenantsError;

  for (const tenant of tenants) {
    // Récupérer les modules nécessitant une action
    const { data: modules } = await supabase
      .from('modules')
      .select('*')
      .eq('tenant_id', tenant.id)
      .in('status', ['degraded', 'active'])
      .gt('error_count', 0);

    if (!modules || modules.length === 0) continue;

    // Obtenir les actions (IA ou règles)
    const actions = await getActions(modules, tenant.id, supabase);

    // Insérer chaque action dans pending_actions
    for (const action of actions) {
      await supabase.from('pending_actions').insert({
        tenant_id: tenant.id,
        module_id: action.moduleId,
        action_type: action.type,
        reason: action.reason,
        status: 'pending',
      });
    }
  }

  return new Response(JSON.stringify({ success: true }), {
    headers: { ...corsHeaders, 'Content-Type': 'application/json' },
  });
});

async function getActions(modules: any[], tenantId: string, supabase: any) {
  // Essayer d'utiliser l'IA si une clé est configurée (tenant ou globale)
  const aiProvider = Deno.env.get('AI_PROVIDER') || 'openai';
  const globalKey = Deno.env.get(`${aiProvider.toUpperCase()}_API_KEY`);

  // Chercher une clé spécifique au tenant
  const { data: tenantSecret } = await supabase
    .from('tenant_secrets')
    .select('encrypted_key, iv, auth_tag')
    .eq('tenant_id', tenantId)
    .eq('provider', aiProvider)
    .single();

  let apiKey: string | null = null;
  if (tenantSecret) {
    // Déchiffrer (à implémenter selon votre méthode de chiffrement)
    apiKey = await decryptSecret(tenantSecret.encrypted_key, tenantSecret.iv, tenantSecret.auth_tag);
  } else if (globalKey) {
    apiKey = globalKey;
  }

  let actions: any[] = [];
  if (apiKey) {
    actions = await callAIProvider(modules, aiProvider, apiKey);
  }
  if (actions.length === 0) {
    actions = ruleBasedDecision(modules);
  }
  return actions;
}

function ruleBasedDecision(modules: any[]): any[] {
  const actions = [];
  for (const mod of modules) {
    if (mod.error_count >= 5) {
      actions.push({ moduleId: mod.id, type: 'restart', reason: 'error_count >= 5' });
    } else if (mod.error_count >= 3) {
      actions.push({ moduleId: mod.id, type: 'repair', reason: 'error_count >= 3' });
    } else if (mod.status === 'degraded') {
      actions.push({ moduleId: mod.id, type: 'rollback', reason: 'degraded status' });
    } else if (mod.error_count > 0 && mod.last_success && new Date(mod.last_success).getTime() < Date.now() - 3600000) {
      actions.push({ moduleId: mod.id, type: 'restart', reason: 'no success for 1 hour' });
    }
  }
  return actions;
}

async function callAIProvider(modules: any[], provider: string, apiKey: string): Promise<any[]> {
  // Implémentation minimale pour OpenAI (à adapter pour Claude, Gemini, DeepSeek)
  if (provider === 'openai') {
    const response = await fetch('https://api.openai.com/v1/chat/completions', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${apiKey}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model: 'gpt-4o-mini',
        messages: [
          {
            role: 'user',
            content: `Analyse ces modules et retourne un JSON avec les actions à prendre (repair, restart, rollback, kill, ignore). Modules: ${JSON.stringify(modules)}`
          }
        ],
        temperature: 0,
      }),
    });
    const data = await response.json();
    try {
      const content = data.choices[0].message.content;
      const parsed = JSON.parse(content);
      return Array.isArray(parsed) ? parsed : [];
    } catch {
      return [];
    }
  }
  // Autres providers à implémenter selon besoin
  return [];
}
