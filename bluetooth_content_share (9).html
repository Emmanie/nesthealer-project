CREATE OR REPLACE FUNCTION process_pending_action()
RETURNS TRIGGER
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
BEGIN
  PERFORM
    edge_function('process-action', jsonb_build_object('record', row_to_json(NEW)));
  RETURN NEW;
END;
$$;

CREATE TRIGGER process_action_trigger
  AFTER INSERT ON pending_actions
  FOR EACH ROW
  EXECUTE FUNCTION process_pending_action();
