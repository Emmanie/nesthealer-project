import { createClient } from 'https://esm.sh/@supabase/supabase-js@2.39.0';

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Methods': 'POST, OPTIONS',
  'Access-Control-Allow-Headers': 'Content-Type, Authorization',
};

Deno.serve(async (req) => {
  if (req.method === 'OPTIONS') return new Response('ok', { headers: corsHeaders });

  const supabase = createClient(
    Deno.env.get('SUPABASE_URL')!,
    Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
  );

  const { data: tenants } = await supabase.from('tenants').select('id, plan, config');
  for (const tenant of tenants || []) {
    const { data: modules } = await supabase
      .from('modules')
      .select('*')
      .eq('tenant_id', tenant.id)
      .in('status', ['degraded', 'active'])
      .gt('error_count', 0);
    if (!modules?.length) continue;

    // Playbooks
    const { data: playbooks } = await supabase
      .from('playbooks')
      .select('*')
      .eq('tenant_id', tenant.id)
      .eq('enabled', true)
      .order('priority', { ascending: false });

    const actions = [];
    for (const mod of modules) {
      let action = null;
      for (const pb of playbooks || []) {
        let match = true;
        for (const [k, v] of Object.entries(pb.conditions)) {
          if (mod[k] != v) { match = false; break; }
        }
        if (match) {
          action = { moduleId: mod.id, type: pb.actions.type, reason: `Playbook: ${pb.name}` };
          break;
        }
      }
      if (!action) {
        // Règle par défaut
        const type = mod.error_count >= 5 ? 'restart' : mod.error_count >= 3 ? 'repair' : 'ignore';
        action = { moduleId: mod.id, type, reason: 'rule engine' };
      }
      if (action.type !== 'ignore') actions.push(action);
    }

    for (const a of actions) {
      await supabase.from('pending_actions').insert({
        tenant_id: tenant.id,
        module_id: a.moduleId,
        action_type: a.type,
        reason: a.reason,
        status: 'pending',
      });
    }
  }
  return new Response(JSON.stringify({ success: true }), { headers: { ...corsHeaders, 'Content-Type': 'application/json' } });
});

import { createClient } from 'https://esm.sh/@supabase/supabase-js@2.39.0';

Deno.serve(async (req) => {
  const { record } = await req.json();
  const supabase = createClient(
    Deno.env.get('SUPABASE_URL')!,
    Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
  );

  const { id, tenant_id, module_id, action_type, reason } = record;
  let success = false;
  try {
    switch (action_type) {
      case 'repair':
      case 'restart':
        const { error } = await supabase
          .from('modules')
          .update({ error_count: 0, status: 'active', last_success: new Date().toISOString() })
          .eq('id', module_id);
        success = !error;
        break;
      case 'rollback':
        const { data: mod } = await supabase.from('modules').select('previous_config, config').eq('id', module_id).single();
        if (mod?.previous_config) {
          const { error } = await supabase
            .from('modules')
            .update({ config: mod.previous_config, previous_config: mod.config, error_count: 0, status: 'active' })
            .eq('id', module_id);
          success = !error;
        }
        break;
      case 'kill':
        const { error: killErr } = await supabase.from('modules').update({ status: 'dead' }).eq('id', module_id);
        success = !killErr;
        break;
    }
    await supabase.from('pending_actions').update({ status: success ? 'completed' : 'failed', completed_at: new Date().toISOString() }).eq('id', id);
    await supabase.from('action_logs').insert({ tenant_id, module_id, action: action_type, reason, success });
  } catch (err) {
    await supabase.from('pending_actions').update({ status: 'failed', error_message: err.message, completed_at: new Date().toISOString() }).eq('id', id);
  }
  return new Response(JSON.stringify({ success: true }));
});

import { createClient } from 'https://esm.sh/@supabase/supabase-js@2.39.0';

Deno.serve(async () => {
  const supabase = createClient(
    Deno.env.get('SUPABASE_URL')!,
    Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
  );

  const { data: tenants } = await supabase.from('tenants').select('id, name');
  for (const tenant of tenants || []) {
    const startDate = new Date(); startDate.setDate(startDate.getDate() - 7);
    const { data: actions } = await supabase
      .from('action_logs')
      .select('*')
      .eq('tenant_id', tenant.id)
      .gte('created_at', startDate.toISOString());
    const total = actions?.length || 0;
    const repaired = actions?.filter(a => a.success).length || 0;
    const mttr = actions?.filter(a => a.success).reduce((acc, a) => acc + (a.details?.duration || 0), 0) / (repaired || 1);

    const html = `
      <h1>Rapport hebdomadaire - ${tenant.name}</h1>
      <p>Incidents : ${total}</p>
      <p>Réparations automatiques : ${repaired}</p>
      <p>Temps moyen de réparation : ${Math.round(mttr)} sec</p>
      <p>👍 L'IA vous a fait gagner ${repaired * 5} minutes cette semaine.</p>
    `;
    // Email via Resend
    await fetch('https://api.resend.com/emails', {
      method: 'POST',
      headers: { 'Authorization': `Bearer ${Deno.env.get('RESEND_API_KEY')}`, 'Content-Type': 'application/json' },
      body: JSON.stringify({
        from: 'noreply@selfhealing.com',
        to: Deno.env.get('DEFAULT_ALERT_EMAIL'),
        subject: 'Rapport hebdomadaire',
        html,
      }),
    }).catch(console.error);
  }
  return new Response('ok');
});

import { useEffect, useState } from 'react';
import { supabase } from '@/lib/supabaseClient';
export default function SiteList({ tenantId }) {
  const [modules, setModules] = useState([]);
  useEffect(() => {
    fetchModules();
    const channel = supabase.channel('modules').on('postgres_changes', { event: '*', schema: 'public', table: 'modules', filter: `tenant_id=eq.${tenantId}` }, fetchModules).subscribe();
    return () => supabase.removeChannel(channel);
  }, [tenantId]);
  async function fetchModules() {
    const { data } = await supabase.from('modules').select('*').eq('tenant_id', tenantId);
    setModules(data || []);
  }
  if (modules.length === 0) return <div className="card text-center text-gray-500">Aucun site</div>;
  return (
    <div className="card overflow-x-auto">
      <table className="min-w-full">

'use client';
import { useEffect, useState } from 'react';
import { supabase } from '@/lib/supabaseClient';
import { LineChart, Line, XAxis, YAxis, CartesianGrid, Tooltip, ResponsiveContainer } from 'recharts';
export default function AnalyticsPage() {
  const [data, setData] = useState([]);
  useEffect(() => {
    supabase.auth.getUser().then(async ({ data: { user } }) => {
      if (user) {
        const { data: tenant } = await supabase.from('tenants').select('id').eq('user_id', user.id).single();
        if (tenant) {
          const { data: logs } = await supabase.from('action_logs').select('created_at, success').eq('tenant_id', tenant.id).order('created_at', { ascending: true });
          const grouped = logs?.reduce((acc, log) => {
            const date = new Date(log.created_at).toLocaleDateString();
            if (!acc[date]) acc[date] = { date, total: 0, repaired: 0 };
            acc[date].total++;
            if (log.success) acc[date].repaired++;
            return acc;
          }, {});
          setData(Object.values(grouped || {}));
        }
      }
    });
  }, []);
  return (
    <div>
      <h1 className="text-2xl font-bold mb-4">Analytique SLA</h1>
      <div className="card">
        <ResponsiveContainer width="100%" height={300}>
          <LineChart data={data}>
            <CartesianGrid strokeDasharray="3 3" />
            <XAxis dataKey="date" />
            <YAxis />
            <Tooltip />
            <Line type="monotone" dataKey="repaired" stroke="#2563eb" />
          </LineChart>
        </ResponsiveContainer>
      </div>
    </div>
  );
}

import { createClient } from '@supabase/supabase-js';
import { NextResponse } from 'next/server';
export async function GET(req, { params }) {
  const supabase = createClient(process.env.NEXT_PUBLIC_SUPABASE_URL, process.env.SUPABASE_SERVICE_ROLE_KEY);
  const { data: tenant } = await supabase.from('tenants').select('plan, custom_limit').eq('id', params.tenantId).single();
  const { count } = await supabase.from('modules').select('*', { count: 'exact', head: true }).eq('tenant_id', params.tenantId);
  let limit = tenant?.plan === 'basic' ? 1 : tenant?.plan === 'premium' ? 3 : tenant?.plan === 'enterprise' ? 10 : 999999;
  if (tenant?.plan === 'custom') limit = tenant.custom_limit || 0;
  return NextResponse.json({ plan: tenant?.plan, limit, used: count || 0 });
}
