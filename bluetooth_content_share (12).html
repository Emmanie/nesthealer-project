// supabase/functions/orchestrate/index.ts
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2.39.0';

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Methods': 'POST, OPTIONS',
  'Access-Control-Allow-Headers': 'Content-Type, Authorization',
};

Deno.serve(async (req) => {
  if (req.method === 'OPTIONS') return new Response('ok', { headers: corsHeaders });

  const supabase = createClient(
    Deno.env.get('SUPABASE_URL')!,
    Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
  );

  // Récupérer tous les tenants actifs
  const { data: tenants } = await supabase
    .from('tenants')
    .select('id, plan, config');

  for (const tenant of tenants) {
    // Récupérer les modules dégradés ou en erreur
    const { data: modules } = await supabase
      .from('modules')
      .select('*')
      .eq('tenant_id', tenant.id)
      .in('status', ['degraded', 'active'])
      .gt('error_count', 0);

    if (!modules || modules.length === 0) continue;

    // Décision IA (appel à un provider ou règle)
    const actions = await decideActions(modules, tenant.id, supabase);

    // Enregistrer les actions dans une table "pending_actions"
    for (const action of actions) {
      await supabase.from('pending_actions').insert({
        tenant_id: tenant.id,
        module_id: action.moduleId,
        action_type: action.type,
        reason: action.reason,
        status: 'pending',
      });
    }
  }

  return new Response(JSON.stringify({ success: true }), {
    headers: { ...corsHeaders, 'Content-Type': 'application/json' },
  });
});

// Moteur de décision (IA ou règles)
async function decideActions(modules: any[], tenantId: string, supabase: any) {
  // Essayer d’abord le provider IA du tenant (si clé fournie)
  const { data: tenantKey } = await supabase
    .from('tenant_secrets')
    .select('provider, encrypted_key, iv, auth_tag')
    .eq('tenant_id', tenantId)
    .eq('provider', Deno.env.get('AI_PROVIDER')!);

  let actions = [];
  if (tenantKey) {
    // Appel à l’API IA (OpenAI, Claude, etc.)
    // Simulé ici par un appel à une autre Edge Function
    actions = await callAIProvider(modules, tenantId, tenantKey);
  }
  if (actions.length === 0) {
    // Fallback sur moteur de règles
    actions = ruleBasedDecision(modules);
  }
  return actions;
}

function ruleBasedDecision(modules: any[]) {
  const actions = [];
  for (const mod of modules) {
    if (mod.error_count >= 5) {
      actions.push({ moduleId: mod.id, type: 'restart', reason: 'error_count >= 5' });
    } else if (mod.error_count >= 3) {
      actions.push({ moduleId: mod.id, type: 'repair', reason: 'error_count >= 3' });
    } else if (mod.status === 'degraded') {
      actions.push({ moduleId: mod.id, type: 'rollback', reason: 'degraded status' });
    }
  }
  return actions;
}
